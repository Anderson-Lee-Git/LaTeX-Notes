\newcommand{\details}[2]{
	{\small
			{\expandafter\ifx\expandafter\relax\detokenize{#1}\relax
					\textnormal{#2}
				\else
					\textnormal{\textbf{#1}: #2}
				\fi
				\vspace{-1pt}}
		}
}

}

\newcommand{\pub}[4]{
	{\small
			{#1,
				``#2''.
				\expandafter\ifx\expandafter\relax\detokenize{#4}\relax
					\textnormal{\emph{#3}}
				\else
					\textnormal{In \emph{#3} }
				\fi
				\expandafter\ifx\expandafter\relax\detokenize{#4}\relax
				\else
					\textnormal{(#4)}
				\fi
				\vspace{-2pt}}
		}
}
