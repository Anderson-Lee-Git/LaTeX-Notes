		\begin{algorithmic}[1]
			\Function{Compute $\#C$ }{$l$}
			\If{$l = 0$}
			\State{\Return{$1$}}
			\EndIf
			\If{$l = 1$}
			\State{\Return{$2$}}
			\EndIf
			\\
			\State{$C[\ ] = \{0, 1\}$}
			\For {$(i=3 \text{ to } l)$}
			\State{$C[i] = C[i - 1] + C[i - 2]$}
			\EndFor
			\State{\Return{$C[l]$}}
			\EndFunction
		\end{algorithmic}
	\end{algorithm}
